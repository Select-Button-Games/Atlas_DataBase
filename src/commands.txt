CREATE DATABASE ComplexDB;
USE ComplexDB;
ADD TABLE Projects (ProjectID INT PRIMARY KEY, ProjectName STRING, StartDate TIMESTAMP, Budget FLOAT);
ADD TABLE Employees (EmpID INT PRIMARY KEY, EmpName STRING, Role STRING, Salary FLOAT, DeptID INT REFERENCES Departments(DeptID));
ADD TABLE Departments (DeptID INT PRIMARY KEY, DeptName STRING);
ADD TABLE ProjectAssignments (AssignmentID INT PRIMARY KEY, ProjectID INT REFERENCES Projects(ProjectID), EmpID INT REFERENCES Employees(EmpID), AssignmentDate TIMESTAMP);
